using System;

class Program
{
    static void Main()
    {
        // 1. Solicitud de datos
        Console.Write("Ingresa tu nombre: ");
        string nombre = Console.ReadLine();

        Console.Write("Ingresa tu apellido: ");
        string apellido = Console.ReadLine();

        Console.Write("Ingresa tu edad: ");
        // Convertimos el texto ingresado a un número entero (int)
        int edad = int.Parse(Console.ReadLine());

        Console.Write("Ingresa tu altura: ");
        // Convertimos el texto ingresado a un número decimal (double)
        double altura = double.Parse(Console.ReadLine());

        // 2. Mostrar la información formateada
        Console.WriteLine("\n------------------------------------");
        Console.WriteLine($"Hola, mi nombre es {nombre} {apellido}.");
        Console.WriteLine($"Tengo {edad} años y mido {altura} metros.");
        Console.WriteLine("------------------------------------");

        // Pausa para que no se cierre la consola inmediatamente
        Console.WriteLine("\nPresiona cualquier tecla para salir...");
        Console.ReadKey();
    }
}
